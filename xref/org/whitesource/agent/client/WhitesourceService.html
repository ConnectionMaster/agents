16  package org.whitesource.agent.client;
17  
18  import org.whitesource.agent.api.dispatch.CheckPoliciesResult;
19  import org.whitesource.agent.api.dispatch.GetInHouseRulesResult;
20  import org.whitesource.agent.api.dispatch.RequestFactory;
21  import org.whitesource.agent.api.dispatch.UpdateInventoryResult;
22  import org.whitesource.agent.api.model.AgentProjectInfo;
23  
24  import java.util.Collection;
25  
26  /**
27   * A facade to the communication layer with the White Source service.
28   *
29   * @author Edo.Shor
30   */
31  public class WhitesourceService {
32  
33      /* --- Members --- */
34  
35      private WssServiceClient client;
36  
37      private RequestFactory requestFactory;
38  
39      /* --- Constructors --- */
40  
41      public WhitesourceService() {
42          this("generic", "1.0");
43      }
44  
45      public WhitesourceService(final String agent, final String agentVersion) {
46          this(agent, agentVersion, null);
47      }
48  
49      public WhitesourceService(final String agent, final String agentVersion, final String serviceUrl) {
50          requestFactory = new RequestFactory(agent, agentVersion);
51  
52          String url = serviceUrl;
53          if (url == null || url.trim().length() == 0) {
54              url = System.getProperty(ClientConstants.SERVICE_URL_KEYWORD, ClientConstants.DEFAULT_SERVICE_URL);
55          }
56          client = new WssServiceClientImpl(url);
57      }
58  
59      /* --- Public methods --- */
60  
61      /**
62       * The method update the White Source organization account with the given OSS information.
63       *
64       * @param orgToken     Organization token uniquely identifying the account at white source..
65       * @param projectInfos OSS usage information to send to white source.
66       * @return Result of updating white source.
67       * @throws WssServiceException In case of errors while updating white source.
68       */
69      public UpdateInventoryResult update(String orgToken,
70                                          String product,
71                                          String productVersion,
72                                          Collection<AgentProjectInfo> projectInfos)
73              throws WssServiceException {
74          return client.updateInventory(
75                  requestFactory.newUpdateInventoryRequest(orgToken, product, productVersion, projectInfos));
76      }
77  
78      /**
79       * The method check the policies application of the given OSS information.
80       *
81       * @param orgToken     Orgnization token uniquely identifying the account at white source..
82       * @param projectInfos OSS usage information to send to white source.
83       * @return Potential result of applying the currently defined policies.
84       * @throws WssServiceException In case of errors while checking the policies with white source.
85       */
86      public CheckPoliciesResult checkPolicies(String orgToken,
87                                               String product,
88                                               String productVersion,
89                                               Collection<AgentProjectInfo> projectInfos)
90              throws WssServiceException {
91          return client.checkPolicies(
92                  requestFactory.newCheckPoliciesRequest(orgToken, product, productVersion, projectInfos));
93      }
94  
95      public GetInHouseRulesResult getInHouseRules(String orgToken) throws WssServiceException {
96          return client.getInHouseRules(requestFactory.newGetInHouseRulesRequest(orgToken));
97      }
98  
99      /**
100      * The method close the underlying client to the White Source service.
101      *
102      * @see org.whitesource.agent.client.WssServiceClient#shutdown()
103      */
104     public void shutdown() {
105         client.shutdown();
106     }
107 
108     /* --- Getters / Setters --- */
109 
110     public WssServiceClient getClient() {
111         return client;
112     }
113 
114     public void setClient(WssServiceClient client) {
115         this.client = client;
116     }
117 
118     public RequestFactory getRequestFactory() {
119         return requestFactory;
120     }
121 
122     public void setRequestFactory(RequestFactory requestFactory) {
123         this.requestFactory = requestFactory;
124     }
125 
126 }
