97      /**
98       * The method generate the policy check report
99       *
100      * @param outputDir Directory where report files will be created.
101      * @param pack      <code>True</code> to create a zip file from the resulting directory.
102      * @param properties Properties for template engine.
103      *
104      * @return  File reference to the resulting report.
105      *
106      * @throws IOException In case of errors during report generation process.
107      */
108     public File generate(File outputDir, boolean pack, Properties properties) throws IOException {
109         if (result == null) {
110             throw new IllegalStateException("Check policies result is null");
111         }
112 
113         // prepare working directory
114         File workDir = new File(outputDir, "whitesource");
115         File reportFile = workDir;
116         if (!workDir.exists() && !workDir.mkdir()) {
117             throw new IOException("Unable to make output directory: " + workDir);
118         }
119 
120         // create actual report
121         VelocityEngine engine = createTemplateEngine(properties);
122         VelocityContext context = createTemplateContext();
123         FileWriter fw = new FileWriter(new File(workDir, "index.html"));
124         try {
125             engine.mergeTemplate(TEMPLATE_FOLDER + TEMPLATE_FILE, "UTF-8", context, fw);
126             fw.flush();
127         } finally {
128             FileUtils.close(fw);
129         }
130 
131         // copy resources
132         copyReportResources(workDir);
133 
134         // package report into a zip archive
135         if (pack) {
136             reportFile = new File(outputDir, "whitesource.zip");
137             ZipOutputStream zos = new ZipOutputStream(new FileOutputStream(reportFile));
138             FileUtils.packDirectory(workDir, zos);
139             FileUtils.deleteRecursive(workDir);
140         }
141 
142         return reportFile;
143     }

147     /**
148      * Create and initialize a template engine to use for generating reports.
149      *
150      * @param properties properties to use for engine initialization. May be null.
151      *
152      * @return A new instance of th template engine to use.
153      */
154     protected VelocityEngine createTemplateEngine(Properties properties) {
155         Properties actualProperties = properties;
156         if (actualProperties == null) {
157             actualProperties = new Properties();
158         }
159 
160         String resourceLoader = actualProperties.getProperty(VelocityEngine.RESOURCE_LOADER);
161         if (StringUtils.isBlank(resourceLoader)) {
162             actualProperties.setProperty(VelocityEngine.RESOURCE_LOADER, "classpath");
163             actualProperties.setProperty("classpath.resource.loader.class", ClasspathResourceLoader.class.getName());
164         }
165 
166         String loggerClass = actualProperties.getProperty(VelocityEngine.RUNTIME_LOG_LOGSYSTEM_CLASS);
167         if (StringUtils.isBlank(loggerClass)) {
168             actualProperties.setProperty(VelocityEngine.RUNTIME_LOG_LOGSYSTEM_CLASS,
169                     "org.apache.velocity.runtime.log.NullLogChute");
170         }
171 
172         VelocityEngine velocity = new VelocityEngine(actualProperties);
173         velocity.init();
174 
175         return velocity;
176     }

178     /**
179      * Create the context holding all the information of the report.
180      */
181     protected VelocityContext createTemplateContext() {
182         VelocityContext context = new VelocityContext();
183 
184         context.put("result", result);
185         context.put("hasRejections", result.hasRejections());
186         context.put("licenses", createLicenseHistogram(result));
187         context.put("creationTime", SimpleDateFormat.getInstance().format(new Date()));
188 
189         if (StringUtils.isNotBlank(buildName)) {
190             context.put("buildName", buildName);
191         }
192 
193         if (StringUtils.isNotBlank(buildNumber)) {
194             context.put("buildNumber", buildNumber);
195         }
196 
197         return context;
198     }

213     private Collection<LicenseHistogramDataPoint> createLicenseHistogram(CheckPoliciesResult result) {
214         Collection<LicenseHistogramDataPoint> dataPoints = new ArrayList<LicenseHistogramDataPoint>();
215 
216         // create distribution histogram
217         Map<String, Integer> licenseHistogram = new HashMap<String, Integer>();
218         for (Map.Entry<String, Collection<ResourceInfo>> entry : result.getProjectNewResources().entrySet()) {
219             for (ResourceInfo resource : entry.getValue()) {
220                 for (String license : resource.getLicenses()) {
221                     licenseHistogram.put(license, MapUtils.getInteger(licenseHistogram, license, 0) + 1);
222                 }
223             }
224         }
225 
226         // sort by count descending
227         List<Map.Entry<String, Integer>> licenses = new ArrayList<Map.Entry<String, Integer>>(licenseHistogram.entrySet());
228         Collections.sort(licenses, new Comparator<Map.Entry<String, Integer>>() {
229 
230             public int compare(Map.Entry<String, Integer> o1, Map.Entry<String, Integer> o2) {
231                 return o2.getValue().compareTo(o1.getValue());
232             }
233         });
234 
235 
236         // create data points
237         if (!licenses.isEmpty()) {
238             // first licenses
239             for (Map.Entry<String, Integer> entry : licenses.subList(0, Math.min(LICENSE_LIMIT, licenses.size()))) {
240                 dataPoints.add(new LicenseHistogramDataPoint(entry.getKey(), entry.getValue()));
241             }
242 
243             // aggregation of histogram tail
244             int tailSize = licenses.size() - LICENSE_LIMIT;
245             int tailSum = 0;
246             if (tailSize > 0) {
247                 for (Map.Entry<String, Integer> entry : licenses.subList(LICENSE_LIMIT, licenses.size())) {
248                     tailSum += entry.getValue();
249                 }
250                 dataPoints.add(new LicenseHistogramDataPoint(OTHER_LICENSE + " (" + tailSize + ")", tailSum));
251             }
252 
253             // normalize bar height
254             float factor = MAX_BAR_HEIGHT / (float) Math.max(tailSum, licenses.get(0).getValue());
255             for (LicenseHistogramDataPoint dataPoint : dataPoints) {
256                 dataPoint.setHeight((int) (factor * dataPoint.getOccurrences()));
257             }
258         }
259 
260         return dataPoints;
261     }
