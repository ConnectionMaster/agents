424     private HttpRequestHandler createHandler(final Object result) {
425         return createHandler(result, ResultEnvelope.STATUS_SUCCESS, ResultEnvelope.MESSAGE_OK);
426     }
427 
428     private HttpRequestHandler createHandler(final Object result, final int status, final String message) {
429         return new HttpRequestHandler() {
430 
431             @Override
432             public void handle(HttpRequest request, HttpResponse response,
433                                HttpContext context) throws HttpException, IOException {
434                 Gson gson = new Gson();
435                 String data = gson.toJson(result);
436                 ResultEnvelope envelope = new ResultEnvelope(status, message, data);
437                 String json = gson.toJson(envelope);
438                 HttpEntity entity = new StringEntity(json, "utf8");
439                 response.setEntity(entity);
440 
441                 response.setStatusCode(HttpStatus.SC_OK);
442             }
443         };
444     }
